# 210918_도커파일 예제_웹서버 php설치
```yaml
FROM ubuntu:18.04
MAINTAINER 951237 <951237@gmail.com>

# 사용자의 입력 받지 않기
ENV DEIAN_FRONTEND = noninteractive

RUN apt-get update
RUN apt-get install -y apache2
RUN apt-get install -y software-properties-common
RUN add-apt-repository ppa:ondrej/php

RUN apt-get update
RUN apt-get install -y php5.6

EXPOSE 80

CMD [“apachect1”, “-D”, “FOREGROUND”]

```

- 빌드 실행하기
```bash
# 빌드실행 명령 
docker build -t example . # 도커파일이 든 폴더에서 실행
```
